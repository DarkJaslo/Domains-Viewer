void MyGLWidget::viewTransformEuler () {
    glm::mat4 VM(1.0f);
    
    //Alejar la escena de la camara
    VM = glm::translate(VM,glm::vec3(0.,0.,-2.0f*radiEscena));
    
    //Rotaciones
    VM = glm::rotate(VM,angle (90),glm::vec3(0,0,1));
    
    
    //Mover toda la escena al 0,0,0
    VM = glm::translate(VM,-centreEscena);
    glUniformMatrix4fv (viewLoc, 1, GL_FALSE, &VM[0][0]);
}
